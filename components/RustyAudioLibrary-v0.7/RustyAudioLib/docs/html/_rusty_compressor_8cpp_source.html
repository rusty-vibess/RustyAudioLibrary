12void RustyCompressor::process(float* sample)
13{
14 float in = *sample;
15
16 // Convert to log dB
17 x_g = 20 * log10f(fabs(in));
18 if(x_g < -120) {
19 x_g = -120;
20 }
21
22 // Gain computation - Calculates how much gain reduction to be introduced
23 // based on whether or not the sample is above the threshold
24 if(2 * (x_g - _threshold) < -_kneeWidth)
25 {
26 y_g = x_g;
27 }
28 else if(2 * abs(x_g - _threshold) <= _kneeWidth)
29 {
30 y_g = x_g + (1 / _ratio - 1) * powf((x_g - _threshold + _kneeWidth / 2), 2) / (2 * _kneeWidth);
31 }
32 else if(2 * (x_g - _threshold) > _kneeWidth)
33 {
34 y_g = _threshold + (x_g - _threshold) / _ratio;
35 }
36
37 // Gain computing summation
38 x_l = x_g - y_g;
39
40 // Peak Corrected Level Detector Gain Smoothing - Smooths the reduction or
41 // increase over time based on Attack/Release values
42 if (x_l > y_l_prev)
43 {
44 y_l_dB = _release * y_l_prev + (1 - _release) * x_l;
45 }
46 else
47 {
48 y_l_dB = _attack * y_l_prev + (1 - _attack) * x_l;
49 }
50
51 // Assigning previous sample for feedback - Feedback detector calculates
52 // sample smoothing based on previous output.
53 y_l_prev = y_l_dB;
54 y_l = powf(10, (y_l_dB / 20)); // Converting to linear amplitude
55
56 // Gain staging
57 cv = y_l *_makeupGain;
58 *sample = in * cv; // Multiplying income sample by compressor control voltage
59}
