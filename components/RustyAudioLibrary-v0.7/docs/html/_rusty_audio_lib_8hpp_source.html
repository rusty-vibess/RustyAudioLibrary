212 void setParam( std::string unit, std::string param, std::string value ) {
213 auto p = _processors.find(unit); // Checking for matching name in _modulators and _processors
214 auto m = _modulators.find(unit);
215
216 if(p == _processors.end() && m == _modulators.end()) { // BAD_CASE no match
217 std::cerr << "BAD_UNIT_NAME `" << unit << "` does not match a known processor" << std::endl;
218 return;
219 }
220 if(p != _processors.end()) // PROC_MATCH processor match case
221 {
222 p->second->setParameter(param, value); // Set relevant param to value
223 }
224 if(m != _modulators.end()) // MODU_MATCH modulator match case
225 {
226 m->second->setParameter(param, value); // Set relevant param to value
227 }
228 }
